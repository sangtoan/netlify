	\hfill
	\begin{enumerate}
		\item Tính khoảng cách từ điểm $M(5;2)$ đến đường thẳng $\Delta\colon 4x-3y-1=0$.
		\item Tính góc giữa hai đường thẳng $\Delta_1\colon x-3 y-2022=0$ và $\Delta_2\colon 2x+4y+2023=0$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Khoảng cách từ điểm $M(5;2)$ đến đường thẳng $\Delta\colon 4x-3y-1=0$ là
	$$\mathrm{d}(M;\Delta)=\dfrac{|4\cdot5-3\cdot2-1|}{\sqrt{4^2+(-3)^2}}=\dfrac{13}{5}.$$
	\item Gọi $\alpha$ là góc giữa hai đường thẳng $\Delta_1\colon x-3 y-2022=0$ và $\Delta_2\colon 2 x+4 y+2023=0$.\\
	Đường thẳng $\Delta_1$ có véc-tơ pháp tuyến $\overrightarrow{n}_{\Delta_1}=(1;-3)$.\\
	Đường thẳng $\Delta_2$ có véc-tơ pháp tuyến $\overrightarrow{n}_{\Delta_2}=(2;4)$.\\
	Ta có $$\cos\alpha=\dfrac{|1\cdot2-3\cdot4|}{\sqrt{1+9}\cdot\sqrt{4+16}}=\dfrac{\sqrt{2}}{2}.$$
	Vậy $\alpha=45^\circ$ hay góc giữa hai đường thẳng $\Delta_1$ và $\Delta_2$ là $45^\circ$.
\end{enumerate}
}

\begin{bt}%[0X3B2-4]%[Dự án đề kiểm tra HKII NH22-23- Nguyễn Văn Sang]%[THPT Lê Lợi - Quảng Trị]

\begin{enumerate}
		\item $3$ viên bi lấy ra thuộc $3$ màu khác nhau.
		\item $3$ viên bi lấy ra có đúng $2$ viên bi màu vàng.
	\end{enumerate}
\loigiai{
	Gọi $\Omega$ là không gian mẫu, ta có $n(\Omega)=\mathrm{C}_{15}^3=455$.
\begin{enumerate}
	\item Gọi $A$ là biến cố ``$3$ viên bi lấy ra thuộc $3$ màu khác nhau.''\\
	Suy ra $n(A)=\mathrm{C}_{6}^1\cdot \mathrm{C}_{5}^1\cdot \mathrm{C}_{4}^1=120$.\\
	Xác suất biến cố $A$ là
