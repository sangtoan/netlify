\subsection{PHÂN TÍCH VÉC-TƠ} \begin{dang}{Phân Tích Vecto} \end{dang} \begin{vd}%[2H2H1-2] Cho hình lăng trụ $ABC.A'B'C'$ có $M$ là trung điểm của $BB'$. Đặt $\overrightarrow{CA}=\overrightarrow{a}$, $\overrightarrow{CB}=\overrightarrow{b}$, $\overrightarrow{CC'}=\overrightarrow{c}$. Chứng minh rằng $\overrightarrow{AM}=\overrightarrow{b}-\overrightarrow{a}+\dfrac{1}{2}\overrightarrow{c}$. \loigiai{ Do $M$ là trung điểm $BB'$ nên ta có $\overrightarrow{AM}=\dfrac{1}{2}\left(\overrightarrow{AB}+\overrightarrow{AB'}\right)$.\\ Do $ABB'A'$ là hình bình hành nên $\overrightarrow{AB'}=\overrightarrow{AB}+\overrightarrow{AA'}$.\\ Suy ra $\overrightarrow{AM}=\dfrac{1}{2}\left(\overrightarrow{AB}+\overrightarrow{AB}+\overrightarrow{AA'}\right)=\overrightarrow{AB}+\dfrac{1}{2}\overrightarrow{AA'}$.\\ Mà $\overrightarrow{AB}=\overrightarrow{CB}-\overrightarrow{CA}=\overrightarrow{b}-\overrightarrow{a}$ và $\overrightarrow{AA'}=\overrightarrow{CC'}=\overrightarrow{c}$\\ nên $\overrightarrow{AM}=\overrightarrow{b}-\overrightarrow{a}+\dfrac{1}{2}\overrightarrow{c}$. } \end{vd} \begin{vd} Cho hình lăng trụ tam giác $ABC.A'B'C'$ có $\overrightarrow{AA'}=\overrightarrow{a}$, $\overrightarrow{AB}=\overrightarrow{b}$, $\overrightarrow{AC}=\overrightarrow{c}$.\\ Chứng minh rằng $\overrightarrow{B'C}=\overrightarrow{c}-\overrightarrow{a}-\overrightarrow{b}$ và $\overrightarrow{BC'}=\overrightarrow{a}-\overrightarrow{b}+\overrightarrow{c}$. \loigiai{ \begin{itemize} \item Ta có $\begin{aligned}[t] \overrightarrow{B'C} &=\overrightarrow{BC}-\overrightarrow{BB'}\\ &=(\overrightarrow{AC}-\overrightarrow{AB})-\overrightarrow{BB'} \\ &=\overrightarrow{AC}-\overrightarrow{BB'}-\overrightarrow{AB}\\ &=\overrightarrow{c}-\overrightarrow{a}-\overrightarrow{b}. \end{aligned}$ \item $\begin{aligned}[t] \overrightarrow{BC'} &=\overrightarrow{BA}+\overrightarrow{AA'}+\overrightarrow{A'C'}\\ &=-\overrightarrow{b}+\overrightarrow{a}+\overrightarrow{c}\\ &=\overrightarrow{a}-\overrightarrow{b}+\overrightarrow{c}. \end{aligned}$ \end{itemize} } \end{vd} \begin{vd}%[Trịnh Văn Xuân]%[1H3K1-3] Cho tứ diện $ABCD$. Gọi $M,N,P,Q$ là các điểm thỏa mãn $\overrightarrow{MA}=-2\overrightarrow{MB}$, $\overrightarrow{NB}=-2\overrightarrow{NC}$, $\overrightarrow{PC}=-2\overrightarrow{PD}$, $\overrightarrow{QD}=-2\overrightarrow{QA}$. Hãy phân tích các vectơ $\overrightarrow{MN},\overrightarrow{MP},\overrightarrow{MQ}$ theo ba vectơ $\overrightarrow{AB},\overrightarrow{AC},\overrightarrow{AD}$. \loigiai{ Ta có $\overrightarrow{MA}=-2\overrightarrow{MB}=-2\left(\overrightarrow{MA}+\overrightarrow{AB}\right)$ suy ra $\overrightarrow{MA}=-\dfrac{2}{3}\overrightarrow{AB}$; \\ $\overrightarrow{NB}=-2\overrightarrow{NC} \Leftrightarrow \overrightarrow{NA}+\overrightarrow{AB}=-2\left(\overrightarrow{NA}+\overrightarrow{AC}\right)$ suy ra $\overrightarrow{NA}=-\dfrac{1}{3}\left(\overrightarrow{AB}+2\overrightarrow{AC}\right)$; \\ $\overrightarrow{PC}=-2\overrightarrow{PD} \Leftrightarrow \overrightarrow{PA}+\overrightarrow{AC}=-2\left(\overrightarrow{PA}+\overrightarrow{AD}\right)$ suy ra $\overrightarrow{PA}=-\dfrac{1}{3}\left(\overrightarrow{AC}+2\overrightarrow{AD}\right)$; \\ $\overrightarrow{QD}=-2\overrightarrow{QA} \Leftrightarrow \overrightarrow{QA}+\overrightarrow{AD}=-2\overrightarrow{QA}$ suy ra $\overrightarrow{QA}=-\dfrac{1}{3}\overrightarrow{AD}$. \\ Từ đó, suy ra \\ $\overrightarrow{MN}=\overrightarrow{MA}-\overrightarrow{NA}=-\dfrac{2}{3}\overrightarrow{AB}+\dfrac{1}{3}\left(\overrightarrow{AB}+2\overrightarrow{AC}\right)=-\dfrac{1}{3}\overrightarrow{AB}+\dfrac{2}{3}\overrightarrow{AC}$; \\ $\overrightarrow{MP}=\overrightarrow{MA}-\overrightarrow{PA}=-\dfrac{2}{3}\overrightarrow{AB}+\dfrac{1}{3}\left(\overrightarrow{AC}+2\overrightarrow{AD}\right)=-\dfrac{2}{3}\overrightarrow{AB}+\dfrac{1}{3}\overrightarrow{AC}+\dfrac{2}{3}\overrightarrow{AD}$; \\ $\overrightarrow{MQ}=\overrightarrow{MA}-\overrightarrow{QA}=-\dfrac{2}{3}\overrightarrow{AB}+\dfrac{1}{3}\overrightarrow{AD}$. } \end{vd}
