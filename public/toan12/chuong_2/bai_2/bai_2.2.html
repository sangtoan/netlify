\subsubsection{Nhóm Tích Vô Hướng - Tính Góc} \begin{ex}%[Nguyễn Trung Kiên, Dự án 12-EX-8-2019]%[2H3Y1-2] Trong không gian tọa độ $Oxyz$, góc giữa hai véc tơ $\overrightarrow{i}$ và $\overrightarrow{u}=(-\sqrt{3};0;1)$ là \choice {$30^\circ$} {$120^\circ$} {$60^\circ$} {\True $150^\circ$} \loigiai {Ta có $\cos\left(\vec{i},\vec{u}\right)=\dfrac{\vec{i}\cdot\vec{u}}{\left|\vec{i}\right|\cdot\left|\vec{u}\right|}=\dfrac{-\sqrt{3}}{1\cdot 2}=-\dfrac{\sqrt{3}}{2}\Rightarrow \left(\vec{i},\vec{u}\right)=150^\circ$.} \end{ex} %%=====Câu 16 \begin{ex}%% Trong không gian $Oxyz$, cho hai vectơ $\overrightarrow u=(1;2;0)$ và $\overrightarrow v=(-1;3;4)$. Tích vô hướng của hai vectơ $\overrightarrow u$ và $\overrightarrow v$ bằng \choice {\True$5$} {$-3$} {$0$} {$-1$} \loigiai{Ta có $\overrightarrow u\cdot\overrightarrow v=-1+6+0=5.$ } \end{ex} %%=====Câu 17 \begin{ex}%% Trong không gian $Oxyz$, cho hai vectơ $\overrightarrow u=(3;m;-1)$ và $\overrightarrow v=(m-1;-2;2)$ vuông góc với nhau. Giá trị của $m$ bằng \choice {$1$} {\True$5$} {$-2$} {$3$} \loigiai{Ta có $\overrightarrow u=(3;m;-1)$ vuông góc $\overrightarrow v=(m-1;-2;2)$ khi và chỉ khi $3(m-1)-2(m)-2=0$ \\$\Leftrightarrow m=5.$ } \end{ex} %%=====Câu 18 \begin{ex}%% Trong không gian $Oxyz$, cho hai vectơ $\overrightarrow u=(2;1;2)$ và $\overrightarrow v=(1;0;2)$. Cosin góc tạo bởi hai vectơ đó bằng \choice {$ \dfrac{1}{\sqrt5} $} {$ \dfrac{2}{3\sqrt5} $} {\True$ \dfrac{2}{\sqrt5} $} {$ 0 $} \loigiai{Ta có $\cos(\overrightarrow u,\overrightarrow v)=\dfrac{2+0+4}{\sqrt{4+1+4}\cdot\sqrt{1+0+4}}=\dfrac{2}{\sqrt5}.$ } \end{ex} %%=====Câu 19 \begin{ex}%% Trong không gian $Oxyz$, cho các điểm $A(2;1;1),B(4;0;1),C(3;1;-2)$. Cosin của $\angle ABC$ bằng \choice {$ \dfrac{-3}{\sqrt{55}} $} {$ \dfrac{1}{2} $} {$ \dfrac{2}{\sqrt{33}} $} {\True$ \dfrac{3}{\sqrt{55}} $} \loigiai{Ta có $\overrightarrow{BA}=(-2;1;0)$ và $\overrightarrow{BC}=(-1;1;-3)$. Do đó $$\cos(\angle ABC)=\cos(\overrightarrow {BA},\overrightarrow {BC})=\dfrac{2+1+0}{\sqrt{4+1+0}\cdot\sqrt{1+1+9}}=\dfrac{3}{\sqrt{55}}.$$ } \end{ex} %%=====Câu 20 \begin{ex}%% Trong không gian $Oxyz$, cho các điểm $A(3;0;1),B(2;3;0),C(1;1;1)$. Sin của $\angle ABC$ bằng \choice {$ \sqrt{\dfrac{6}{11}} $} {$ \sqrt{\dfrac{2}{3}} $} {$ \sqrt{\dfrac{1}{3}} $} {\True$ \sqrt{\dfrac{5}{11}} $} \loigiai{Ta có $\overrightarrow{BA}=(1;-3;1)$ và $\overrightarrow{BC}=(-1;-2;1)$. Do đó $$\cos(\angle ABC)=\cos(\overrightarrow {BA},\overrightarrow {BC})=\dfrac{-1+6+1}{\sqrt{1+9+1}\cdot\sqrt{1+4+1}}=\sqrt{\dfrac{6}{11}}\Rightarrow \sin(\angle ABC)=\sqrt{\dfrac{5}{11}}.$$ } \end{ex} %%=====Câu 21 \begin{ex}%% Trong không gian $Oxyz$, cho các điểm $A(2;1;0),B(0;-1;2),C(2;-1;-1),D(3;3;1)$. Cosin của góc tạo bởi của hai đường thẳng $AB,CD$ bằng \choice {\True$ \sqrt{\dfrac{1}{7}} $} {$ -\sqrt{\dfrac{1}{7}} $} {$ \sqrt{\dfrac{6}{7}} $} {$ \sqrt{\dfrac{7}{9}} $} \loigiai{Ta có $\overrightarrow{AB}=(-2;-2;2)$ và $\overrightarrow{CD}=(1;4;2)$. Do đó $$\cos(AB,CD)=\left| \cos(\overrightarrow {AB},\overrightarrow {CD})\right| =\left| \dfrac{-2-8+4}{\sqrt{4+4+4}\cdot\sqrt{1+16+4}}\right| =\sqrt{\dfrac{1}{7}}.$$ } \end{ex}
