\section{TOẠ ĐỘ OXYZ} \subsection{Phần Trắc Nghiệm} \begin{ex} Cho điểm $M$ thỏa mãn $\overrightarrow{OM} = 2\vec{i} + \vec{j}$. Tọa độ của điểm $M$ là \choice {$M(0; 2; 1)$} {$M(1; 2; 0)$} {$M(2; 0; 1)$} {\True $M(2; 1; 0)$} \loigiai{ \textbf{Phương pháp:} \\ $\overrightarrow{OM} = x\vec{i} + y\vec{j} + z\vec{k} \Rightarrow M(x; y; z)$ \\ $\overrightarrow{OM} = 2\vec{i} + \vec{j} = 2\vec{i} + 1\vec{j} + 0\vec{k} \Rightarrow M(2; 1; 0)$ \\ } \end{ex} \begin{ex} Cho hai điểm $A(-1; 2; -3)$ và $B(2; -1; 0)$. Tọa độ của vectơ $\overrightarrow{AB}$ là \choice {$\overrightarrow{AB} = (1; -1; 1)$} {$\overrightarrow{AB} = (3; 3; -3)$} {$\overrightarrow{AB} = (1; 1; -3)$} {\True $\overrightarrow{AB} = (3; -3; 3)$} \loigiai{ \textbf{Phương pháp:} Cho $A(x_A; y_A; z_A)$, $B(x_B; y_B; z_B)$. Khi đó, \\ $\overrightarrow{AB} = (x_B - x_A; y_B - y_A; z_B - z_A)$ \\ Ta có $\overrightarrow{AB} = (2 + 1; -1 - 2; 0 + 3) = (3; -3; 3)$. \\ } \end{ex} \begin{ex} Cho hai điểm $A(3; -2; 3)$ và $B(-1; 2; 5)$. Tọa độ trung điểm $I$ của đoạn thẳng $AB$ là \choice {$I(-2; 2; 1)$} {\True $I(1; 0; 4)$} {$I(2; 0; 8)$} {$I(2; -2; -1)$} \loigiai{ \textbf{Phương pháp:} Cho $A(x_A; y_A; z_A)$, $B(x_B; y_B; z_B)$. Khi đó, \\ $I(x_I; y_I; z_I)$ là trung điểm $AB \Rightarrow$ \[ \begin{cases} x_I = \dfrac{x_A + x_B}{2} \\ y_I = \dfrac{y_A + y_B}{2} \\ z_I = \dfrac{z_A + z_B}{2} \end{cases} \] Tọa độ trung điểm $I$ là $I\left( \dfrac{3 + (-1)}{2}, \dfrac{-2 + 2}{2}, \dfrac{3 + 5}{2} \right)$ hay $I(1; 0; 4)$. \\ } \end{ex} \begin{ex} Cho ba điểm $A(1; 3; 5)$, $B(2; 0; 1)$, $C(0; 9; 0)$. Tọa độ trọng tâm $G$ của tam giác $ABC$ là \choice {$G(3; 12; 6)$} {$G(1; 5; 2)$} {$G(1; 0; 5)$} {\True $G(1; 4; 2)$} \loigiai{ \textbf{Phương pháp:} Cho $A(x_A; y_A; z_A)$, $B(x_B; y_B; z_B)$, $C(x_C; y_C; z_C)$. Khi đó, \\ $G(x_G; y_G; z_G)$ là trọng tâm $\triangle ABC \Rightarrow$ \[ \begin{cases} x_G = \dfrac{x_A + x_B + x_C}{3} \\ y_G = \dfrac{y_A + y_B + y_C}{3} \\ z_G = \dfrac{z_A + z_B + z_C}{3} \end{cases} \] Tọa độ trọng tâm $G$ của tam giác $ABC$ là $G\left( \dfrac{1 + 2 + 0}{3}; \dfrac{3 + 0 + 9}{3}; \dfrac{5 + 1 + 0}{3} \right)$ hay $G(1; 4; 2)$. \\ } \end{ex} \begin{ex} Cho $A(1; 2; -1)$, $B(2; 1; -3)$, $C(-3; 5; 1)$. Điểm $D$ sao cho $ABCD$ là hình bình hành có tọa độ là \choice {\True $D(-4; 6; 3)$} {$D(-2; 2; 5)$} {$D(-2; 8; -3)$} {$D(-4; 6; -5)$} \loigiai{ Gọi $D(x_D; y_D; z_D)$, ta có: \[ \overrightarrow{DC} = (-3 - x_D; 5 - y_D; z_D - 1) \] \[ \overrightarrow{AB} = (1; -1; -2) \] Theo đề ta có $ABCD$ là hình bình hành $\Leftrightarrow \overrightarrow{AB} = \overrightarrow{DC} \Leftrightarrow$ \[ \begin{cases} -3 - x_D = 1 \\ 5 - y_D = -1 \\ z_D - 1 = -2 \end{cases} \Leftrightarrow \begin{cases} x_D = -4 \\ y_D = 6 \\ z_D = 3 \end{cases} \] Vậy $D(-4; 6; 3)$. } \end{ex} \begin{ex} Gọi $\alpha$ là góc giữa hai vectơ $\vec{u} = (0; -1; 0)$ và $\vec{v} = \left(\sqrt{3}; 1; 0\right)$. Giá trị của $\alpha$ là \choice {$\alpha = \dfrac{\pi}{6}$} {$\alpha = \dfrac{\pi}{3}$} {\True $\alpha = \dfrac{2\pi}{3}$} {$\alpha = \dfrac{\pi}{2}$} \loigiai{ Ta có: $\cos\alpha = \dfrac{\vec{u} \cdot \vec{v}}{|\vec{u}| \cdot |\vec{v}|} = \dfrac{0 \cdot \sqrt{3} + (-1) \cdot 1 + 0 \cdot 0}{\sqrt{0^2 + (-1)^2 + 0^2} \cdot \sqrt{\left(\sqrt{3}\right)^2 + 1^2 + 0^2}} = -\dfrac{1}{2}$. \[ \Rightarrow \alpha = \dfrac{2\pi}{3} \] } \end{ex} \begin{ex} Cho $A(2; -1; 1)$, $B(-1; 3; -1)$, $C(5; -3; 4)$. Tích vô hướng $\overrightarrow{AB} \cdot \overrightarrow{BC}$ có giá trị là \choice {$48$} {$-48$} {$52$} {\True $-52$} \loigiai{ \textbf{Phương pháp:} Cho $\overrightarrow{a} = (a_1; a_2; a_3)$, $\overrightarrow{b} = (b_1; b_2; b_3)$. Ta có: \[ \overrightarrow{a} \cdot \overrightarrow{b} = a_1 b_1 + a_2 b_2 + a_3 b_3 \] Ta có: $\overrightarrow{AB} = (-3; 4; -2)$, $\overrightarrow{BC} = (6; -6; 5)$. \\ \[ \Rightarrow \overrightarrow{AB} \cdot \overrightarrow{BC} = (-3) \cdot 6 + 4 \cdot (-6) + (-2) \cdot 5 = -52 \] Vậy $\overrightarrow{AB} \cdot \overrightarrow{BC} = -52$. } \end{ex} \begin{ex} Cho hai điểm $A(-1; 2; 3)$, $B(1; 0; 2)$. Tọa độ điểm $M$ thỏa mãn $\overrightarrow{AB} = 2\overrightarrow{MA}$ là \choice {\True $M(-2; 3; \dfrac{7}{2})$} {$M(-2; -3; \dfrac{7}{2})$} {$M(-2; 3; 7)$} {$M(-4; 6; 7)$} \loigiai{ Gọi $M(x; y; z)$.
