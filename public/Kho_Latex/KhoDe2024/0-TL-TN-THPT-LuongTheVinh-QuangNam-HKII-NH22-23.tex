		\begin{listEX}

			\item Viết phương trình đường tròn $(C)$ có đường kính $AB$.
		\end{listEX}
		\loigiai{
			\begin{listEX}
				\item Đường thẳng $AB$ đi qua $A(-1;2)$ và $B(3;-4)$ nên nhận $\vec{AB}=(4;-6)$ làm véc-tơ chỉ phương.\\
				Suy ra $AB$ nhận $\vec{n}=(3;2)$ làm một véc-tơ pháp tuyến.\\
				Phương trình tổng quát của $AB\colon 3(x+1)+2(y-2)=0 \Leftrightarrow 3x+2y-1=0$.
				\item Gọi $I$ là trung điểm của $AB$ thì $I(1;-1)$ là tâm của đường tròn $(C)$ đường kính $AB$.\\
				Và $(C)$ có bán kính $R=IB=\sqrt{(3-1)^2+(-4+1)^2}=\sqrt{13}$.\\
				Phương trình đường tròn đường kính $AB$ là $(C)\colon (x-1)^2+(y+1)^2=13$.

	\begin{bt}%[0D2G2-7]%[Dự án đề kiểm tra HKII NH22-23- Dương Phước Sang]%[THPT Lương Thế Vinh - Quảng Nam]
		Từ các chữ số của tập $A=\big\{1;2;3;4;5;6;7\big\}$ có thể lập được bao nhiêu số tự nhiên có $8$ chữ số sao cho trong số có $8$ chữ số của số được lập ra mỗi chữ số của tập $A$ đều có mặt ít nhất một lần và không có hai chữ số chẵn nào đứng cạnh nhau.
		\loigiai{
			Tập $A$ có $7$ phần tử, để lập được số có $8$ chữ số mà các chữ số của $A$ đều xuất hiện ta phải dùng một chữ số $x$ nào đó trong $A$ $2$ lần.
			\begin{itemize}
				\item Trường hợp $1$: $x$ lẻ, tức là $x \in \big\{1;3;5;7\big\}$. Có $4$ cách chọn $x$. \\
				Để không có $2$ chữ số chẵn nào kề nhau ta làm như sau:\\
				$\bullet$ Trước hết, xếp $5$ chữ số $x$, $1$, $3$, $5$, $7$ thành hàng ngang, có $\dfrac{5!}{2!}$ cách xếp.\\
				\phantom{$\bullet$} Có $4$ vị trí nằm giữa mỗi cặp hai chữ số lẻ kề nhau được tạo ra.\\
				$\bullet$ Xếp $3$ chữ số $2$, $4$, $6$ vào $3$ trong $4$ vị trí nêu trên, có $\mathrm{A}_4^3$ cách xếp.\\
				Đến đây, dãy các chữ số đã xếp được hình thành nên một số thỏa mãn yêu cầu đề bài.\\
				Trường hợp $1$ ta lập được $4\cdot \left(\dfrac{5!}{2!}\cdot \mathrm{A}_4^3\right)=5760$ số như thế. 
				\item Trường hợp $2$: $x$ chẵn, tức là $x \in \big\{2;4;6\big\}$. Có $3$ cách chọn $x$. \\
				Để không có $2$ chữ số chẵn nào kề nhau ta chỉ cần xếp $4$ chữ số lẻ ($1$, $3$, $5$, $7$) và $4$ chữ số chẵn ($x$, $2$, $4$, $6$) xen kẽ nhau. Có $2 \cdot 4! \cdot \dfrac{4!}{2!}$ cách xếp chúng xen kẽ.\\
				Như vậy, trường hợp $2$ ta lập được $3\cdot \left(2 \cdot 4! \cdot \dfrac{4!}{2!}\right)=1728$ số thỏa mãn yêu cầu đề bài.
			\end{itemize}
			Vậy tổng cộng có $4\cdot \left(\dfrac{5!}{2!}\cdot \mathrm{A}_4^3\right)+3\cdot \left(2 \cdot 4! \cdot \dfrac{4!}{2!}\right)=5760+1728=7488$ số thỏa mãn yêu cầu đề bài.
		}
	\end{bt}
